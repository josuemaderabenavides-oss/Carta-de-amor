<script>
  // Copiar carta
  const copyBtn = document.getElementById('copyBtn');
  const letterEl = document.getElementById('letter');
  copyBtn.addEventListener('click', async () => {
    try{
      await navigator.clipboard.writeText(letterEl.textContent.trim());
      copyBtn.textContent = '¡Copiado!';
      setTimeout(()=>copyBtn.textContent='Copiar carta',1200);
    }catch(e){
      alert('No se pudo copiar. Selecciona y copia manualmente.');
    }
  });

  // Descargar carta como .txt
  document.getElementById('downloadBtn').addEventListener('click', () => {
    const text = letterEl.textContent.trim();
    const blob = new Blob([text], {type: 'text/plain;charset=utf-8'});
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url; a.download = 'carta_para_mi_amor.txt';
    document.body.appendChild(a); a.click();
    a.remove(); URL.revokeObjectURL(url);
  });

  // Editar carta en sitio (simple)
  document.getElementById('editBtn').addEventListener('click', () => {
    const current = letterEl.textContent.trim();
    const newText = prompt('Edita tu carta:', current);
    if(newText !== null){
      letterEl.textContent = newText;
    }
  });

  // pequeña mejora: guarda en localStorage
  const STORAGE_KEY = 'mi_carta_para_ella_v1';
  // cargar
  const saved = localStorage.getItem(STORAGE_KEY);
  if(saved) letterEl.textContent = saved;
  // guardar cada vez que el usuario edite (detectamos cambios con el botón editar)
  document.getElementById('editBtn').addEventListener('click', ()=> {
    setTimeout(()=> localStorage.setItem(STORAGE_KEY, letterEl.textContent.trim()), 600);
  });
</script>
